### 完整代码（逐行注释版）

```python
# -*- coding: utf-8 -*-                         # 声明文件编码为 UTF-8，支持中文
"""
SQL 安全校验器                                 # 模块文档字符串，描述文件用途

在 LLM 生成的 SQL 执行之前进行多层安全检查：
1. 危险关键字拦截（DROP/DELETE/UPDATE/ALTER/TRUNCATE/INSERT/CREATE）
2. 表名白名单校验（只允许查询元数据配置中声明的表）
3. 基础 SQL 注入特征检测
"""

import re                                        # re = Regular Expression(正则表达式)模块
# 正则表达式用来做"文本模式匹配"——判断字符串是否符合某种模式
# 例如：判断 SQL 中是否包含 "DROP" 这个词

from typing import List, Optional                # typing 模块提供"类型注解"工具
# List[str]   = 表示"字符串列表"这种类型
# Optional[X] = 表示"X 或 None"（可选类型）


class SQLSafetyValidator:                        # class = 定义类，类名用大驼峰命名
    """SQL 安全校验器——在 SQL 送交数据库执行前的最后一道防线"""

    # ── 类变量（class variable）─────────────────────────────
    # 类变量定义在 class 体内、所有方法之外
    # 通过"类名.变量名"访问，如 SQLSafetyValidator.FORBIDDEN_KEYWORDS
    # 所有实例共享同一份类变量

    # 第一层：危险关键字黑名单
    FORBIDDEN_KEYWORDS: List[str] = [
        "DROP",       # 删除表/数据库，如 DROP TABLE users
        "DELETE",     # 删除数据行，如 DELETE FROM orders
        "UPDATE",     # 修改数据，如 UPDATE products SET price=0
        "ALTER",      # 修改表结构，如 ALTER TABLE users ADD age INT
        "TRUNCATE",   # 清空表（比 DELETE 更快，不可回滚）
        "INSERT",     # 插入数据，如 INSERT INTO users VALUES(...)
        "CREATE",     # 创建表/索引，如 CREATE TABLE backup(...)
        "REPLACE",    # 替换数据（MySQL 特有，先删后插）
        "GRANT",      # 授权，如 GRANT SELECT ON ...
        "REVOKE",     # 撤销权限
        "RENAME",     # 重命名，如 RENAME TABLE old TO new
        "LOAD",       # 加载外部文件，如 LOAD DATA INFILE
        "IMPORT",     # 导入数据
    ]

    # 第二层：表名白名单
    ALLOWED_TABLES: List[str] = [
        "dim_region",    # 地区维度表
        "dim_customer",  # 客户维度表
        "dim_product",   # 商品维度表
        "dim_date",      # 时间维度表
        "fact_order",    # 订单事实表
    ]

    # 第三层：SQL 注入特征检测
    INJECTION_PATTERNS: List[str] = [
        r"'\s*OR\s+'1'\s*=\s*'1",     # ' OR '1'='1  → 经典注入，永远为真
        r"'\s*OR\s+1\s*=\s*1\s*--",    # ' OR 1=1 --   → 注入变体
        r"UNION\s+SELECT",              # UNION SELECT   → 合并其他表数据
        r";\s*DROP\s+",                 # ;DROP          → 多语句注入
        r";\s*DELETE\s+",               # ;DELETE        → 多语句注入
        r";\s*UPDATE\s+",               # ;UPDATE        → 多语句注入
        r"--\s*$",                      # --             → 行尾注释截断
        r"/\*.*\*/",                    # /*...*/        → 块注释绕过
    ]

    @classmethod
    # @classmethod = 装饰器，让方法变成"类方法"
    # 类方法：不用创建对象，直接用"类名.方法名()"调用
    # 普通方法要先写 obj = ClassName()，再 obj.method()
    # 类方法直接写 ClassName.method()，参数中有 cls 代替 self
    def validate(
        cls,                 # cls = 类本身（普通方法是 self = 实例）
        sql: str,            # 参数类型注解：sql 应该是字符串
        allowed_tables: Optional[List[str]] = None,
        # Optional[List[str]] = 可选参数，可以是字符串列表或 None
        # = None = 默认值为 None（调用时不传这个参数也没关系）
    ) -> str:               # 返回值类型注解：返回一个字符串
        """
        对 SQL 执行多层安全检查

        Args:
            sql: LLM 生成的 SQL 语句
            allowed_tables: 额外的允许表名（可选）

        Returns:
            str: 清洗后的 SQL

        Raises:
            ValueError: 校验不通过时抛出，附带具体原因
        """

        # ── 空值检查 ──
        # if not sql = Python 的"假值"判断
        # None、空字符串 ""、0、空列表 [] 都算 False
        if not sql or not sql.strip():
            raise ValueError("SQL 语句为空，拒绝执行")
            # raise = 抛出异常，中止函数执行
            # ValueError = Python 内置异常类型，表示"值不对"

        # .upper() = 字符串方法，全部转大写
        # 例如 "select * from users" → "SELECT * FROM USERS"
        # 为什么要转大写？因为后面的正则匹配要在忽略大小写的 SQL 中找关键字
        sql_upper = sql.upper()

        # 三元表达式：值1 if 条件 else 值2
        tables = allowed_tables if allowed_tables is not None else cls.ALLOWED_TABLES

        # ═══ 第一层：危险关键字拦截 ═══
        for keyword in cls.FORBIDDEN_KEYWORDS:
            # for 循环：每次从列表取一个关键字赋给 keyword

            # 拼接正则表达式：\b + 关键字 + \b
            # r"..." = 原始字符串，里面的 \ 不会被转义
            # \b = 正则的"单词边界"元字符
            # 例如 r"\bDROP\b" 匹配 "DROP TABLE" 的 DROP
            # 但不匹配 "DROPDOWN" 的 DROP（它是单词的一部分）
            pattern = r"\b" + keyword + r"\b"

            # re.search(pattern, text) = 在 text 中搜索 pattern
            # 找到返回 Match 对象（真值），找不到返回 None（假值）
            if re.search(pattern, sql_upper):
                # 被拦截！构造错误信息
                start_idx = max(0, sql_upper.index(keyword) - 20)
                # .index() = 找子串的位置，返回索引（从0开始）
                # max(0, ...) = 防止索引为负数
                end_idx = sql_upper.index(keyword) + len(keyword) + 20
                # len() = 内置函数，返回长度
                context = sql_upper[start_idx:end_idx]
                # [start:end] = 切片，取字符串的一个片段

                raise ValueError(
                    f"SQL 安全拦截：检测到危险关键字 '{keyword}'。"
                    # f"..." = f-string，花括号内的是变量
                    f"问数系统只允许 SELECT 查询，禁止修改数据库。"
                    f"被拦截的 SQL 片段: ...{context}..."
                )

        # ═══ 第二层：只允许 SELECT/WITH 查询 ═══
        stripped = sql_upper.strip()
        # .strip() = 去掉首尾空白字符

        # .startswith("SELECT") = 检查字符串是否以 "SELECT" 开头
        # and = 逻辑与；not = 逻辑非
        # WITH 开头也允许，因为 CTE(公用表表达式) 是这样的：
        #   WITH temp AS (SELECT ...) SELECT * FROM temp
        if not stripped.startswith("SELECT") and not stripped.startswith("WITH"):
            raise ValueError(
                f"SQL 安全拦截：只允许 SELECT/WITH 开头的查询语句。"
                f"当前 SQL 以 '{stripped[:50]}...' 开头"
                # stripped[:50] = 切片，取前 50 个字符
            )

        # ═══ 第三层：SQL 注入特征检测 ═══
        for pattern in cls.INJECTION_PATTERNS:
            # re.IGNORECASE = 不区分大小写标志
            # 因为注入可能用混合大小写绕过简单检测
            if re.search(pattern, sql, re.IGNORECASE):
                raise ValueError(
                    f"SQL 安全拦截：检测到可疑的注入特征。"
                    f"这可能是一次 SQL 注入攻击，已拒绝执行。"
                )

        # 全部检查通过 → 返回清理后的 SQL
        return sql.strip()
```

### 目的
在 SQL 执行前设置三层安全检查：
1. **关键字黑名单**：禁止任何写入/修改操作
2. **语句类型检查**：只允许 SELECT 和 WITH 查询
3. **注入特征检测**：识别常见 SQL 注入手法

### 涉及 Python 知识点

| 知识点 | 对应语法 | 在代码中的位置 |
|--------|----------|---------------|
| 类定义 | `class Name:` | 第 11 行 |
| 类变量 | 类体内的变量（在方法外） | FORBIDDEN_KEYWORDS 等 |
| 类方法 | `@classmethod` + `def func(cls):` | 第 37 行 |
| 类型注解 | `: str`, `-> str`, `Optional[List[str]]` | 参数和返回值 |
| 正则表达式 | `re.search(pattern, text)` | 安全检测逻辑 |
| f-string | `f"变量={变量}"` | 错误信息构造 |
| 切片 | `[start:end]` | 截取 SQL 片段 |
| 异常处理 | `raise ValueError(...)` | 拦截时抛出异常 |

```python
def _clean_sql(sql: str) -> str:
    """
    清理 LLM 生成的 SQL 中可能混入的非 SQL 内容
    
    为什么需要这个函数？
    LLM 常把 SQL 包在 Markdown 代码块中：```sql ... ```
    直接执行带标记的文本，MySQL 会报语法错误。
    
    正则表达式解释：
        r"^```(?:sql)?\s*\n?"
        ^         = 行首
        ```       = 三个反引号
        (?:sql)?  = "sql" 可选（非捕获组）
        \s*       = 零个或多个空白
        \n?       = 零个或一个换行
    """
    sql = re.sub(
        r"^```(?:sql)?\s*\n?",      # 匹配开头的代码块标记
        "",                           # 替换为空（即删除）
        sql,                          # 要处理的文本
        flags=re.IGNORECASE           # 不区分大小写
    )
    sql = re.sub(
        r"\n?```\s*$",               # 匹配末尾的代码块标记
        "",                           # 替换为空
        sql                           # 要处理的文本
    )
    return sql.strip()
```

```python
async def run_sql(state, runtime):
    writer = runtime.stream_writer
    step = "执行SQL"
    writer({"type": "progress", "step": step, "status": "running"})

    try:
        # Step 1: 取出并清理 SQL
        sql = _clean_sql(state["sql"])
        logger.info(f"[SQL安全] 待执行的 SQL: {sql[:200]}...")

        # Step 2: ⭐ 新增——三层安全校验
        try:                                          # 内层 try：专门捕获安全校验错误
            sql = SQLSafetyValidator.validate(sql)     # 调用防火墙
            logger.info("[SQL安全] 安全校验通过 ✅")
        except ValueError as safety_err:               # 捕获 ValueError
            logger.warning(f"[SQL安全] 校验拦截: {safety_err}")
            writer({"type": "progress", "step": step, "status": "error"})
            writer({"type": "error", "message": str(safety_err)})
            return                                     # 优雅终止，不抛异常

        # Step 3: 执行 SQL（原逻辑）
        dw_mysql_repository = runtime.context["dw_mysql_repository"]
        result = await dw_mysql_repository.run(sql)

        # Step 4: 返回结果
        logger.info(f"[SQL安全] 执行成功，返回 {len(result)} 行")
        writer({"type": "progress", "step": step, "status": "success"})
        writer({"type": "result", "data": result})

    except Exception as e:                            # 外层 except：捕获执行错误
        logger.error(f"[SQL安全] 执行失败: {e}")
        writer({"type": "progress", "step": step, "status": "error"})
        writer({"type": "error", "message": str(e)})
        raise
```

### 核心设计：嵌套 try/except

```
外层 try ←── 捕获数据库执行错误（表不存在、字段错误等）
    │
    内层 try ←── 捕获安全校验错误（危险关键字、注入特征等）
        │
        SQLSafetyValidator.validate(sql)
        │
        ├── 通过 → 继续执行 SQL
        └── 失败 → return（优雅退出，不触发外层 except）
```

### 目的
在执行 SQL 之前插入安全校验步骤，保护数据库不被 LLM 的恶意或错误输出破坏。

### 涉及 Python 知识点

| 知识点 | 语法 | 代码位置 |
|--------|------|----------|
| 嵌套 try/except | 内层 try 在外层 try 内 | run_sql 函数体 |
| 异常抑制 | `return` 替代 `raise` | 安全拦截分支 |
| await | `await db.run(sql)` | SQL 执行处 |
| 切片 | `sql[:200]` | 日志输出 |

        self._embed_url = f"{base_url}/embed"

        """用 aiohttp 发 POST 到 TEI 的 /embed 端点"""
        async with aiohttp.ClientSession() as session:
            async with session.post(self._embed_url, json={"inputs": texts}) as resp:
                return await resp.json()
    
    async def aembed_query(self, text):
        embeddings = await self.aembed_documents([text])
        return embeddings[0]

class EmbeddingClientManager:
    def __init__(self, config):
        self.client: Optional[TEIEmbeddings] = None  # 类型改为自定义类
    
    def init(self):
        self.client = TEIEmbeddings(base_url=self._get_url())  # ✅ 用自定义实现
```

### 目的
`langchain-huggingface` 从 1.2.x 版本开始，`HuggingFaceEndpointEmbeddings` 的 `model` 参数必须是一个 HuggingFace Hub 的仓库名（如 `BAAI/bge-large-zh-v1.5`），不再接受 `http://localhost:8081` 这样的 URL。

改为自定义 `TEIEmbeddings` 类，通过 `aiohttp` 直接调用 TEI 服务的 HTTP API。

### 文件 #7：`app/repositories/mysql/dw/dw_mysql_repository.py` — 数据仓库增删改查

#### 完整代码

```python
"""
数仓 MySQL 仓储

这一层对应文档里的 DW Repository，职责是到真实数仓中补齐配置文件里
没有显式维护的信息，例如字段类型和字段示例值。Service 层只关心
"需要哪些信息"，具体怎样查数仓由仓储层统一封装
SQL 生成闭环中的数据库环境读取 SQL 校验和最终查询执行也集中放在这里
"""

from sqlalchemy import text
# ↑ text() = SQLAlchemy 的"文本 SQL"函数
#   用来执行原生 SQL 字符串，而不是用 ORM 的链式查询
#   因为 LLM 生成的就是文本 SQL，所以这里必须用 text()

from sqlalchemy.ext.asyncio import AsyncSession
# ↑ AsyncSession = SQLAlchemy 的异步数据库会话
#   所有数据库操作都要通过 Session 来执行
#   async 版本支持 await，不阻塞事件循环


class DWMySQLRepository:
    """负责查询数仓真实表结构和字段样例值"""

    def __init__(self, session: AsyncSession):
        # 构造函数：接收一个数据库会话对象并保存
        # 后续所有方法都用这个 session 来执行 SQL
        self.session = session

    async def get_column_types(self, table_name: str) -> dict[str, str]:
        """
        查询整张表的字段类型，作为 ColumnInfo.type 的真实来源
        
        Returns:
            dict[str, str] = 字典，key是字段名，value是类型
            例如 {"order_id": "int", "order_amount": "decimal(10,2)"}
        """
        sql = f"show columns from {table_name}"
        # SHOW COLUMNS = MySQL 命令，查看表的所有字段及其类型
        result = await self.session.execute(text(sql))
        # execute() = 执行 SQL
        # text() = 把字符串包装成 SQLAlchemy 能识别的 SQL 对象
        # await = 等待异步数据库操作完成
        
        result_dict = result.mappings().fetchall()
        # mappings() = 把每行结果转成"字典式"访问（像 dict）
        # fetchall() = 取出所有结果行
        
        return {row["Field"]: row["Type"] for row in result_dict}
        # 字典推导式：for row in result_dict 遍历每一行
        # row["Field"] 是字段名，row["Type"] 是字段类型

    async def get_column_values(
        self, table_name: str, column_name: str, limit: int = 10
    ) -> list:
        """
        抽样查询字段示例值，供元数据入库和后续检索链路复用
        
        Returns:
            list = 示例值列表，如 ["华东", "华南", "华北"]
        """
        sql = f"select distinct {column_name} from {table_name} limit {limit}"
        # DISTINCT = 去重，避免返回重复值
        result = await self.session.execute(text(sql))
        return [row[0] for row in result.fetchall()]
        # 列表推导式：取每行的第一个字段值

    async def get_db_info(self):
        """
        读取当前数仓数据库的方言和版本，供 SQL 生成提示词使用
        
        Returns:
            dict = {"dialect": "mysql", "version": "8.0.xx"}
        """
        sql = "select version()"
        result = await self.session.execute(text(sql))
        version = result.scalar()
        # scalar() = 取单个标量值（只有一行一列时用）

        dialect = self.session.bind.dialect.name
        # self.session.bind = 当前 Session 绑定的数据库 Engine
        # .dialect.name = 数据库方言名称，如 "mysql"
        return {"dialect": dialect, "version": version}

    async def validate(self, sql: str):
        """
        ⚠️ 只检查 SQL 语法，不检查语义！
        
        用 EXPLAIN 让数据库提前解析 SQL
        EXPLAIN 会检查：
          - SQL 语法是否正确
          - 表名是否存在
          - 字段名是否存在
        但不检查：
          - 字段是否放在了正确的表上
          - JOIN 条件是否合理
          - 是否会产生笛卡尔积
        """
        sql = f"explain {sql}"
        await self.session.execute(text(sql))

    async def run(self, sql: str) -> list[dict]:
        """
        ⚠️ 直接执行 SQL，没有任何安全检查！
        
        这就是安全防火墙要保护的目标函数。
        LLM 生成的任何 SQL 都会通过这个方法执行。

        Returns:
            list[dict] = 查询结果，每行是一个字典
            如 [{"地区": "华东", "销售额": 100000}, ...]
        """
        result = await self.session.execute(text(sql))
        # text(sql) = 把 LLM 生成的字符串包装成 SQLAlchemy SQL 对象
        # 注意：这里没有做任何过滤！
        
        return [dict(row) for row in result.mappings().fetchall()]
        # mappings() = 每行转成可字典访问的对象
        # dict(row) = 转成真正的 Python 字典
        # 最终返回列表套字典的格式：list[dict]
```

#### 关键方法速查

| 方法 | 做了什么 | 谁调用 |
|------|----------|--------|
| `get_column_types` | 查表的字段类型 | 元数据构建 |
| `get_column_values` | 抽字段的示例值 | 元数据构建 |
| `get_db_info` | 读取数据库版本 | SQL 生成时做提示词 |
| `validate` | EXPLAIN 语法检查 | `validate_sql` 节点 |
| `run` ⚠️ | **直接执行 SQL** | `run_sql` 节点 |

#### 安全风险

`run()` 方法是整个系统的"最危险函数"：
- 如果 LLM 生成 `DROP TABLE`，它就会执行 `DROP TABLE`
- 如果 LLM 生成 `DELETE`，它就会执行 `DELETE`
- 没有任何权限控制或操作验证

这就是为什么我们需要 `SQLSafetyValidator` —— 在 `run()` 被调用之前先过一遍安检。

"""
元数据库 MySQL 仓储

这一层对应文档里的 Meta Repository，负责接收业务实体并落到 Meta MySQL
Repository 自身只关心"如何写入"，而"哪些写操作要放在同一笔事务里"，
由 Service 层统一决定
"""

from sqlalchemy import text
from sqlalchemy.ext.asyncio import AsyncSession

from app.entities.column_info import ColumnInfo
from app.entities.column_metric import ColumnMetric
from app.entities.metric_info import MetricInfo
from app.entities.table_info import TableInfo
# ↑ Entity = 业务实体类，纯 Python 对象，不依赖数据库
#   例如 ColumnInfo(id="c001", name="region_name", type="varchar")
#   实体和 ORM 模型的区别：
#     - 实体：纯业务逻辑，不关心数据库
#     - 模型：绑定数据库表，ORM 框架管理

from app.models.column_info import ColumnInfoMySQL
from app.models.table_info import TableInfoMySQL
# ↑ Model = SQLAlchemy ORM 模型，与数据库表一一对应
#   例如 ColumnInfoMySQL 对应 MySQL 的 column_info 表

from app.repositories.mysql.meta.mappers.column_info_mapper import ColumnInfoMapper
from app.repositories.mysql.meta.mappers.column_metric_mapper import ColumnMetricMapper
from app.repositories.mysql.meta.mappers.metric_info_mapper import MetricInfoMapper
from app.repositories.mysql.meta.mappers.table_info_mapper import TableInfoMapper
# ↑ Mapper = 转换器，负责"实体 ↔ 模型"双向转换
#   to_model(实体) → ORM 模型（用于写入数据库）
#   to_entity(模型) → 业务实体（用于业务逻辑）


class MetaMySQLRepository:
    """负责把元数据业务实体持久化到 Meta MySQL"""

    def __init__(self, session: AsyncSession):
        self.session = session

    # ═══════════════ 写入操作（CUD） ═══════════════

    def save_table_infos(self, table_infos: list[TableInfo]):
        """
        批量保存表元数据
        
        输入是业务实体 TableInfo，通过 Mapper 转成 ORM 模型再保存
        add_all() = 批量标记为"待保存"，不会立刻写入数据库
        实际写入在 session.commit() 时由 Service 层统一触发
        """
        self.session.add_all(
            # 列表推导式：把每个实体转成 ORM 模型
            [TableInfoMapper.to_model(table_info) for table_info in table_infos]
        )

    def save_column_infos(self, column_infos: list[ColumnInfo]):
        """批量保存字段元数据"""
        self.session.add_all(
            [ColumnInfoMapper.to_model(column_info) for column_info in column_infos]
        )

    def save_metric_infos(self, metric_infos: list[MetricInfo]):
        """批量保存指标元数据"""
        self.session.add_all(
            [MetricInfoMapper.to_model(metric_info) for metric_info in metric_infos]
        )

    def save_column_metrics(self, column_metrics: list[ColumnMetric]):
        """批量保存字段与指标的关联关系"""
        self.session.add_all(
            [
                ColumnMetricMapper.to_model(column_metric)
                for column_metric in column_metrics
            ]
        )

    # ═══════════════ 读取操作（R） ═══════════════

    async def get_column_info_by_id(self, id: str) -> ColumnInfo | None:
        """
        按字段 id 查询字段元数据
        
        ColumnInfo | None = 返回类型注解
          竖线 | 表示"或"：返回 ColumnInfo 对象或 None
        
        self.session.get(Model, id) = SQLAlchemy 的主键查询
          相当于 SELECT * FROM column_info WHERE id = ?
        """
        column_info: ColumnInfoMySQL | None = await self.session.get(
            ColumnInfoMySQL, id
        )
        if column_info:
            # 找到了 → 把 ORM 模型转回业务实体
            return ColumnInfoMapper.to_entity(column_info)
        else:
            return None

    async def get_table_info_by_id(self, id: str) -> TableInfo | None:
        """按表 id 查询表元数据"""
        table_info: TableInfoMySQL | None = await self.session.get(TableInfoMySQL, id)
        if table_info:
            return TableInfoMapper.to_entity(table_info)
        else:
            return None

    async def get_key_columns_by_table_id(self, table_id: str) -> list[ColumnInfo]:
        """
        查询指定表的主外键字段，避免 Join 关键字段被向量召回漏掉
        
        为什么需要这个方法？
        向量召回（Qdrant）靠语义匹配，可能漏掉主键/外键这类"技术字段"
        但生成 JOIN SQL 时必须要知道主外键关系
        所以这里直接从 MySQL 查，不走向量检索
        """
        sql = "select * from column_info where table_id = :table_id and role in ('primary_key','foreign_key')"
        # :table_id = SQLAlchemy text SQL 的参数占位符
        # role in ('primary_key','foreign_key') = 只查主键和外键字段
        
        result = await self.session.execute(
            text(sql),                    # 原生 SQL
            {"table_id": table_id}        # 参数绑定（防 SQL 注入）
        )
        # 注意：这里的参数用字典传入，SQLAlchemy 会自动做参数绑定
        # 这和 LLM 生成的 SQL 直接拼接完全不同！
        
        return [ColumnInfo(**dict(row)) for row in result.mappings().fetchall()]
        # **dict(row) = 字典解包，相当于 ColumnInfo(id="...", name="...", ...)
```

"""
MySQL 客户端管理器

统一创建和管理项目中的异步 MySQL 客户端，当前项目会同时连接两套 MySQL
一套是保存结构化元数据的 meta 数据库，一套是模拟教学数仓的 dw 数据库

from sqlalchemy.ext.asyncio import (

from app.conf.app_config import DBConfig, app_config


class MySQLClientManager:
    """管理 MySQL Engine 和 Session 工厂"""

    def __init__(self, config: DBConfig):

        # AsyncEngine | None = 可以是 Engine 或 None（未初始化时）

        # session_factory = 一个函数，每次调用返回一个新的 AsyncSession
        self.config = config

    def _get_url(self) -> str:
        """
        拼接 MySQL 异步连接地址
        
        URL 格式：mysql+驱动名://用户名:密码@主机:端口/数据库名?参数
        
        mysql+aiomysql 的含义：
          mysql         = 数据库类型
          +aiomysql     = 使用 aiomysql 作为异步驱动
          (相当于"用 aiomysql 这个库来连接 MySQL")
        """
        return (
            f"mysql+aiomysql://{self.config.user}:{self.config.password}"
            f"@{self.config.host}:{self.config.port}/{self.config.database}"
            f"?charset=utf8mb4"  # utf8mb4 = 支持 emoji 的 UTF-8
        )

    def init(self):

        Engine = 数据库连接池管理者
        Session = 一次数据库"对话"（事务单位）
        
        Engine 是重量级的（创建一次，全局复用）
        Session 是轻量级的（每次查询创建新的）
        """
        self.engine = create_async_engine(
            self._get_url(),
            pool_size=10,        # 连接池最多保持 10 个连接
            pool_pre_ping=True,  # 每次使用前检查连接是否有效
        )
        self.session_factory = async_sessionmaker(
            self.engine,
            autoflush=True,          # 自动同步内存变更到数据库
            expire_on_commit=False,  # 提交后不使对象过期
        )

    async def close(self):
        """释放连接池资源（程序退出时调用）"""
        await self.engine.dispose()

### 文件 #10：`app/repositories/qdrant/column_qdrant_repository.py` — 向量检索

#### 完整代码

```python
"""
字段向量仓储

管理字段向量集合并把已经准备好的 point 批量写入 Qdrant
Qdrant = 向量数据库，专门做"语义相似度搜索"
"""

from qdrant_client import AsyncQdrantClient
from qdrant_client.http.models import PointStruct
# PointStruct = Qdrant 的数据结构，包含：
#   id       = 唯一标识
#   vector   = 向量（一串浮点数，如 [0.1, -0.3, 0.8, ...]）
#   payload  = 附加数据（如字段名、描述等）

from qdrant_client.models import Distance, VectorParams
# Distance.COSINE = 余弦距离（衡量两个向量"方向"有多接近）
# VectorParams(size=1024) = 向量维度为 1024（对应 bge-large-zh-v1.5 的输出）

from app.conf.app_config import app_config
from app.entities.column_info import ColumnInfo


class ColumnQdrantRepository:
    """负责字段向量集合的创建、写入和基础检索"""

    collection_name = "column_info_collection"
    # collection = Qdrant 的"表"概念，类似 MySQL 的 table

    def __init__(self, client: AsyncQdrantClient):
        self.client = client

    async def ensure_collection(self):
        """
        确保字段向量集合存在
        
        类似于 MySQL 的 CREATE TABLE IF NOT EXISTS
        """
        if not await self.client.collection_exists(self.collection_name):
            await self.client.create_collection(
                collection_name=self.collection_name,
                vectors_config=VectorParams(
                    size=app_config.qdrant.embedding_size,  # 1024 维
                    distance=Distance.COSINE,  # 用余弦距离衡量相似度
                ),
            )

    async def upsert(
        self,
        ids: list[str],              # 字段 ID 列表
        embeddings: list[list[float]],  # 向量列表（每个是 1024 个浮点数）
        payloads: list[dict],         # 字段元数据列表
        batch_size: int = 10,        # 每批写入 10 条
    ):
        """
        分批写入字段向量点
        
        upsert = update + insert：如果同名 ID 已存在就更新，不存在就插入
        """
        # zip() = 把三个列表"拉链"在一起
        # 例如 zip([1,2], ["a","b"], [10,20]) → (1,"a",10), (2,"b",20)
        points: list[PointStruct] = [
            PointStruct(id=id, vector=embedding, payload=payload)
            for id, embedding, payload in zip(ids, embeddings, payloads)
        ]

        # 分批写入，避免一次发太多数据
        for i in range(0, len(points), batch_size):
            # range(0, 100, 10) = [0, 10, 20, 30, ...]
            # points[i : i + batch_size] = 切片取当前批
            await self.client.upsert(
                collection_name=self.collection_name,
                points=points[i : i + batch_size],
            )

    async def search(
        self,
        embedding: list[float],       # 用户问题的向量表示
        score_threshold: float = 0.6, # 最低相似度阈值（0~1）
        limit: int = 20,              # 最多返回 20 条
    ) -> list[ColumnInfo]:
        """
        按向量相似度检索字段元数据
        
        这就是"语义搜索"的核心：
        把用户的问题转成向量 → 在 Qdrant 中找最相似的字段向量
        
        例如：
          用户问"销售额" → 向量化 → 找到"order_amount"字段（语义相近）
        """
        result = await self.client.query_points(
            collection_name=self.collection_name,
            query=embedding,           # 用这个向量去搜
            limit=limit,               # 最多返回多少条
            score_threshold=score_threshold,  # 低于此分的不返回
        )
        # result.points = 匹配的向量点列表
        # point.payload = 之前存入的字段元数据
        # **point.payload = 字典解包创建 ColumnInfo 对象
        return [ColumnInfo(**point.payload) for point in result.points]
```

字段取值 ES 仓储

把字段真实取值组织成 Elasticsearch 全文索引
ES = Elasticsearch，专门做"关键词全文搜索"

                "type": "text",             # text = 全文索引，会分词
                "analyzer": "ik_max_word",  # IK 中文分词器（最细粒度）
                "search_analyzer": "ik_max_word",
            },
            "column_id": {"type": "keyword"},
        },
    }

    def __init__(self, client: AsyncElasticsearch):
        self.client = client

    async def ensure_index(self):
        """确保字段取值索引已经创建好"""
        if not await self.client.indices.exists(index=self.index_name):
            await self.client.indices.create(
                index=self.index_name,
                mappings=self.index_mappings,
            )

    async def index(self, value_infos: list[ValueInfo], batch_size=20):
        """
        分批写入字段取值到 ES 全文索引
        
        例如把 dim_region 表的 "华东"、"华南"、"华北" 写入索引
        这样用户搜"华东"时就能找到对应的字段
        """
        if not value_infos:
            return

        for i in range(0, len(value_infos), batch_size):
            batch_value_infos = value_infos[i : i + batch_size]
            batch_operations = []
            for value_info in batch_value_infos:
                # bulk API 格式：一行 action + 一行 data
                batch_operations.append(
                    {"index": {"_index": self.index_name, "_id": value_info.id}}
                )
                batch_operations.append(asdict(value_info))
            await self.client.bulk(operations=batch_operations)

    async def search(
        self, keyword: str, score_threshold: float = 0.6, limit: int = 20
    ) -> list[ValueInfo]:
        """
        按关键词全文检索字段取值
        
        例如：
          用户问"华东地区的销售额"
          → 提取关键词"华东"
          → 在 ES 中搜索 value="华东"
          → 找到 dim_region.province 字段包含"华东"
        """
        resp = await self.client.search(
            index=self.index_name,
            query={"match": {"value": keyword}},  # match = 在 value 字段中全文搜索
            size=limit,
            min_score=score_threshold,  # 过滤掉相关度太低的
        )
        # _source = ES 文档的原始 JSON 数据
        return [ValueInfo(**hit["_source"]) for hit in resp["hits"]["hits"]]
```
